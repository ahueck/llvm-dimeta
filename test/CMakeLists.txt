macro(pythonize_bool truth_var var)
  if(${truth_var})
    set(${var} True)
  else()
    set(${var} False)
  endif()
endmacro()

function(configure_dimeta_lit_site_cfg input output)
  set(LIT_SITE_CFG_IN_HEADER
      "## Autogenerated for dimeta from ${input}\n## Do not edit!"
  )
  
  set(DIMETA_PROJECT_DIR ${PROJECT_SOURCE_DIR})
  set(DIMETA_LIBRARY_DIR ${CMAKE_BINARY_DIR}/lib)
  set(DIMETA_PASS_DIR ${DIMETA_LIBRARY_DIR}/pass)
  set(DIMETA_TEST_PASS_DIR ${CMAKE_CURRENT_BINARY_DIR}/verifier)
  
  set(DIMETA_SCRIPT_DIR ${CMAKE_BINARY_DIR}/scripts)

  if(${LLVM_VERSION_MAJOR} VERSION_GREATER_EQUAL "13")
    set(DIMETA_OPT_ARGS "-enable-new-pm=0")
  endif()

  dimeta_target_generate_file(${input} ${output})
endfunction()

function(dimeta_add_lit_testsuite target comment)
  cmake_parse_arguments(ARG "" "" "PARAMS;DEPENDS;ARGS" ${ARGN})

  foreach(param ${ARG_PARAMS})
    list(APPEND TEST_PARAMS --param ${param})
  endforeach()

  add_custom_target(${target}
    COMMAND ${DIMETA_LIT_EXEC} ${ARG_ARGS} ${TEST_PARAMS} ${ARG_UNPARSED_ARGUMENTS}
    COMMENT "${comment}"
    USES_TERMINAL
  )

  if (ARG_DEPENDS)
    add_dependencies(${target} ${ARG_DEPENDS})
  endif()
endfunction()

function(dimeta_add_lit_target)
  cmake_parse_arguments(ARG "" "" "SUITES" ${ARGN})

  foreach(suite IN LISTS ARG_SUITES)
    if("${suite}" STREQUAL "all")
      set(SUITE_PATH ${CMAKE_CURRENT_SOURCE_DIR})
      set(TARGET_NAME check-dimeta)
    else()
      set(SUITE_PATH ${CMAKE_CURRENT_SOURCE_DIR}/${suite})
      string(REPLACE "/" "_" suite_target ${suite})
      set(TARGET_NAME check-dimeta-${suite_target})
    endif()

    if(NOT EXISTS ${SUITE_PATH})
      message(WARNING "Could not find suitable lit test target at ${SUITE_PATH}")
      continue()
    endif()

    dimeta_add_lit_testsuite(${TARGET_NAME}
      "Running the lit suite dimeta::${suite}"
      ${SUITE_PATH}
      ARGS -v -j 1
      PARAMS dimeta_site_config=${CMAKE_CURRENT_BINARY_DIR}/lit.site.cfg
      DEPENDS ${DIMETA_TEST_DEPENDS}
    )
  endforeach()
endfunction()
